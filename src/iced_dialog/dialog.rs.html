14/// A message dialog.
15///
16/// Only the content is required, [`buttons`] and the [`title`] are optional.
17///
18/// [`buttons`]: Dialog::with_buttons
19/// [`title`]: Dialog::title
20pub struct Dialog<
21    'a,
22    Message,
23    Theme = iced_widget::Theme,
24    Renderer = iced_widget::Renderer,
25> where
26    Renderer: 'a + core::text::Renderer,
27    Theme: 'a + Catalog,
28{
29    is_open: bool,
30    base: Element<'a, Message, Theme, Renderer>,
31    title: Option<Fragment<'a>>,
32    content: Element<'a, Message, Theme, Renderer>,
33    buttons: Vec<Element<'a, Message, Theme, Renderer>>,
34    font: Option<Renderer::Font>,
35    width: Length,
36    height: Length,
37    spacing: f32,
38    padding: Padding,
39    button_alignment: Alignment,
40    class: <Theme as Catalog>::Class<'a>,
41    title_class: <Theme as text::Catalog>::Class<'a>,
42    container_class: <Theme as container::Catalog>::Class<'a>,
43}

51    /// Creates a new [`Dialog`] with the given base and dialog content.
52    pub fn new(
53        is_open: bool,
54        base: impl Into<Element<'a, Message, Theme, Renderer>>,
55        content: impl Into<Element<'a, Message, Theme, Renderer>>,
56    ) -> Self {
57        Self::with_buttons(is_open, base, content, Vec::new())
58    }
59
60    /// Creates a new [`Dialog`] with the given base, dialog content and buttons.
61    pub fn with_buttons(
62        is_open: bool,
63        base: impl Into<Element<'a, Message, Theme, Renderer>>,
64        content: impl Into<Element<'a, Message, Theme, Renderer>>,
65        buttons: Vec<Element<'a, Message, Theme, Renderer>>,
66    ) -> Self {
67        Self {
68            is_open,
69            base: base.into(),
70            title: None,
71            content: content.into(),
72            buttons,
73            font: None,
74            width: 400.into(),
75            height: 260.into(),
76            spacing: 8.0,
77            padding: 24.into(),
78            button_alignment: Alignment::Start,
79            class: <Theme as Catalog>::default(),
80            title_class: <Theme as Catalog>::default_title(),
81            container_class: <Theme as Catalog>::default_container(),
82        }
83    }

138    /// Adds a button to the [`Dialog`], if `Some`.
139    pub fn push_button_maybe(
140        self,
141        button: Option<impl Into<Element<'a, Message, Theme, Renderer>>>,
142    ) -> Self {
143        if let Some(button) = button {
144            self.push_button(button)
145        } else {
146            self
147        }
148    }
149
150    /// Extends the [`Dialog`] with the given buttons.
151    pub fn extend_buttons(
152        self,
153        buttons: impl IntoIterator<Item = Element<'a, Message, Theme, Renderer>>,
154    ) -> Self {
155        buttons.into_iter().fold(self, Self::push_button)
156    }

158    /// Sets the backdrop color of the [`Dialog`].
159    pub fn backdrop(self, color: impl Into<Color>) -> Self

161        <Theme as Catalog>::Class<'a>: From<StyleFn<'a, Theme>>,

163        let backdrop_color = color.into();
164

194        mut self,

196    ) -> Self
197    where
198        <Theme as container::Catalog>::Class<'a>:

202            (Box::new(style) as container::StyleFn<'a, Theme>).into();

206    /// Sets the style class of the [`Dialog`].
207    #[must_use]

213        self
214    }

216    /// Sets the style class of the [`Dialog`]'s title.
217    #[must_use]
218    pub fn title_class(
219        mut self,
220        class: impl Into<<Theme as text::Catalog>::Class<'a>>,

222        self.title_class = class.into();
223        self
224    }
225
226    /// Sets the style class of the [`Dialog`]'s container.
227    #[must_use]
228    pub fn container_class(
229        mut self,
230        class: impl Into<<Theme as container::Catalog>::Class<'a>>,
231    ) -> Self {
232        self.container_class = class.into();
233        self
234    }
235
236    fn view(self) -> Element<'a, Message, Theme, Renderer>
237    where
238        <Theme as container::Catalog>::Class<'a>:
239            From<container::StyleFn<'a, Theme>>,
240    {
241        if self.is_open {
242            let contents = Container::new(
243                Column::new()
244                    .push_maybe(self.title.map(|title| {
245                        let text = text(title)
246                            .size(20)
247                            .line_height(text::LineHeight::Absolute(Pixels(
248                                26.0,
249                            )))
250                            .class(self.title_class);
251
252                        if let Some(font) = self.font {
253                            text.font(font)
254                        } else {
255                            text
256                        }
257                    }))
258                    .push(vertical_space().height(12))
259                    .push(self.content),
260            )
261            .width(Length::Fill)
262            .padding(self.padding);
263
264            let buttons = Container::new(
265                Row::with_children(self.buttons).spacing(self.spacing),
266            )
267            .height(80)
268            .padding(self.padding);
269
270            let dialog = Container::new(
271                Column::new()
272                    .push(contents)
273                    .push(vertical_space())
274                    .push(buttons),
275            )
276            .width(self.width)
277            .height(self.height)
278            .class(self.container_class)
279            .clip(true);
280
281            modal(self.base, dialog, self.class)
282        } else {
283            self.base
284        }
285    }
286}

288impl<'a, Message, Theme, Renderer> From<Dialog<'a, Message, Theme, Renderer>>
289    for Element<'a, Message, Theme, Renderer>
290where
291    Renderer: 'a + core::Renderer + core::text::Renderer,
292    Theme: 'a + Catalog,
293    Message: 'a + Clone,
294    <Theme as container::Catalog>::Class<'a>:
295        From<container::StyleFn<'a, Theme>>,
296{
297    fn from(dialog: Dialog<'a, Message, Theme, Renderer>) -> Self {
298        dialog.view()
299    }
300}
301
302fn modal<'a, Message, Theme, Renderer>(
303    base: impl Into<Element<'a, Message, Theme, Renderer>>,
304    content: impl Into<Element<'a, Message, Theme, Renderer>>,
305    class: <Theme as Catalog>::Class<'a>,
306) -> Element<'a, Message, Theme, Renderer>
307where
308    Message: 'a + Clone,
309    Renderer: 'a + core::Renderer,
310    Theme: 'a + container::Catalog + Catalog,
311    <Theme as container::Catalog>::Class<'a>:
312        From<container::StyleFn<'a, Theme>>,
313{
314    let area = mouse_area(center(opaque(content)).style(move |theme| {
315        container::Style {
316            background: Some(
317                Catalog::style(theme, &class).backdrop_color.into(),
318            ),
319            ..Default::default()
320        }
321    }));
322
323    stack![base.into(), opaque(area)].into()
324}

326/// The style of a [`Dialog`].
327#[derive(Debug, Clone, Copy, PartialEq)]
328pub struct Style {
329    /// The [`Dialog`]'s backdrop.
330    pub backdrop_color: Color,
331}
332
333/// The theme catalog of a [`Dialog`].
334pub trait Catalog: text::Catalog + container::Catalog {
335    /// The item class of the [`Catalog`].
336    type Class<'a>;
337
338    /// The default class produced by the [`Catalog`].
339    fn default<'a>() -> <Self as Catalog>::Class<'a>;
340
341    /// The default class for the [`Dialog`]'s title.
342    fn default_title<'a>() -> <Self as text::Catalog>::Class<'a> {
343        <Self as text::Catalog>::default()
344    }
345
346    /// The default class for the [`Dialog`]'s container.
347    fn default_container<'a>() -> <Self as container::Catalog>::Class<'a> {
348        <Self as container::Catalog>::default()
349    }
350
351    /// The [`Style`] of a class.
352    fn style(&self, class: &<Self as Catalog>::Class<'_>) -> Style;
353}
354
355/// A styling function for a [`Dialog`].
356pub type StyleFn<'a, Theme> = Box<dyn Fn(&Theme) -> Style + 'a>;
357
358impl Catalog for Theme {
359    type Class<'a> = StyleFn<'a, Self>;
360
361    fn default<'a>() -> <Self as Catalog>::Class<'a> {
362        Box::new(default)
363    }
364
365    fn default_container<'a>() -> <Self as container::Catalog>::Class<'a> {
366        Box::new(|theme| {
367            container::background(
368                theme.extended_palette().background.base.color,
369            )
370        })
371    }
372
373    fn style(&self, class: &<Self as Catalog>::Class<'_>) -> Style {
374        class(self)
375    }
376}
377
378/// The default style of a [`Dialog`].
379pub fn default(_theme: &Theme) -> Style {
380    Style {
381        backdrop_color: color!(0x000000, 0.3),
382    }
